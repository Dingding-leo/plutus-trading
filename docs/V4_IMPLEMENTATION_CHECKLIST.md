    """Return correlation vs BTCUSDT from CORRELATION_MATRIX. Defaults to 0.5 (moderate)."""

    """Return beta vs BTCUSDT from BETA_MATRIX. Defaults to 1.0."""

    """Return V4 tier name: 'MAJOR' | 'LARGE' | 'MID' | 'SMALL'."""
    n = normalize_symbol(symbol)

    if n in TIER_LARGE: return "LARGE"
    if n in TIER_MID:   return "MID"
    return "SMALL"

def calculate_crypto_beta_exposure(
    positions: list[dict],
    equity: float,
) -> float:
    """
    Calculate β-weighted portfolio exposure (from V4_PORTFOLIO_MATRIX.md).

    heat = Σ(|position_value_i| × |β_i|) / equity

    Returns heat float.
    """
    total = 0.0
    for pos in positions:
        value = abs(pos.get("value", 0.0))
        beta = abs(get_beta(pos.get("symbol", "")))
        total += value * beta
    return total / equity if equity > 0 else 0.0
